make - Make build elementÂ¶

This is a BuildElement implementation for using GNU make based build.

Note

The make element is available since format version 9

Here is the default configuration for the make element in full:

# make default configurations

variables:
  make: make PREFIX="%{prefix}"
  make-install: make -j1 PREFIX="%{prefix}" DESTDIR="%{install-root}" install

  # Set this if the sources cannot handle parallelization.
  #
  # notparallel: True

config:

  # Commands for building the software
  #
  build-commands:
  - |
    %{make}

  # Commands for installing the software into a
  # destination folder
  #
  install-commands:
  - |
    %{make-install}

  # Commands for stripping debugging information out of
  # installed binaries
  #
  strip-commands:
  - |
    %{strip-binaries}

# Use max-jobs CPUs for building and enable verbosity
environment:
  MAKEFLAGS: -j%{max-jobs}
  V: 1

# And dont consider MAKEFLAGS or V as something which may
# effect build output.
environment-nocache:
- MAKEFLAGS
- V
